#include<avr/io.h>
void main()
{int j=0,p,d[]={0b00111111,0b00000110,0b01011011,0b01001111,0b01100110,0b01101101,0b01111101,0b00000111,0b01111111,0b01101111};
DDRA=0b00000000;
DDRB=0b11111111;
while(1)
{
if((PINA&0b00000001)==0b00000000)
{while((PINA&0b00000001)==0b00000000);
if(j==10)
j=0;
PORTB=d[j++];
p=j-1;}
if((PINA&0b00000010)==0b00000000)
{while((PINA&0b00000010)==0b00000000);
if(p==0)
p=10;
PORTB=d[--p];
j=p+1;}
}
}
